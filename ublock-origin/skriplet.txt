cepn.js application/javascript
(() => {
    const path = "{{1}}";
    const urlp = location.pathname.toLowerCase();
    if (path === "" || path === "{{1}}" || urlp.indexOf(path) === -1) return;
    const prev = "{{2}}";
    if (prev === "" || prev === "{{2}}") return;
    const next = "{{3}}";
    if (next === "" || next === "{{3}}") return;
    const click = (selector) => {
        document.querySelector(selector).click();
    };
    document.addEventListener(
        "keyup",
        (e) => {
            if (e.altKey && e.key === "b") {
                click(prev);
            } else if (e.altKey && e.key === "n") {

            }
