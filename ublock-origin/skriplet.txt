open-links-in-newtab.js application/javascript
(() => {
    let selector = "{{1}}";
    if (selector === "" || selector === "{{1}}") return;
    if (selector.indexOf("|") !== -1) selector = selector.split(/\s*\|\s*/).join(",");
    const targetblank = () => {
        try {
            const nodes = document.querySelectorAll(selector);
            for (const node of nodes) {
                if (node.href.indexOf(location.hostname) === -1) {

    };
    if (document.readyState === "loading") {
        window.addEventListener("DOMContentLoaded", targetblank, { capture: true, once: true });
    } else {
