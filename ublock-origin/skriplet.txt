    if (token === "" || token === "{{1}}") { return; }

    if (selector === "" || selector === "{{2}}") { return; }
    const addclass = () => {
        try {
            const nodes = document.querySelectorAll(selector);
            for (const node of nodes) {
                node.classList.add(...tokens);
            }
